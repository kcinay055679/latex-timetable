\ExplSyntaxOn{}
\str_new:N \g_plannedHoursTimespanColor
\str_new:N \g_doneHoursTimespanColor
\str_new:N \g_taskTextColor
\str_new:N \g_timespanTextColor

\cs_gset:Npn \getTimespanColor #1{
  \str_case:nn {#1}
    {
      {currentPlannedHours}{\g_plannedHoursTimespanColor}
      {sprintPlannedHours}{\g_plannedHoursTimespanColor}

      {currentDoneHours}{\g_doneHoursTimespanColor}
      {sprintDoneHours}{\g_doneHoursTimespanColor}
    }
}

\cs_gset:Npn \getTaskTextColor {
    \g_taskTextColor
}

\cs_gset:Npn \getTimeSpanTextColor #1{
    \str_case:nn {#1}
    {
      {currentPlannedHours}{000000}
      {sprintPlannedHours}{000000}

      {currentDoneHours}{FFFFFF}
      {sprintDoneHours}{FFFFFF}
    }
}


\newcommand{\setColor}[3]{
    \str_gset:Nn \g_plannedHoursTimespanColor{#1}
    \str_gset:Nn \g_doneHoursTimespanColor{#2}
    \str_gset:Nn \g_taskTextColor{#2}
    \str_gset:Nn \g_timespanTextColor{#2}
}
